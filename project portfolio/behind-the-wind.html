About the project:

Game Engine: Unity

Category: Solo Game

Published: itch.io

Game Was Made: 22.06.2021

Control:

Left Mouse Button - Interact,

W/S/D/A - Move,

Left Shift - Run,

Space - Jump,

Mouse - Look Around,

F - Flashlight,

Wall Run - W/S/D/A + Space.


Description of the game:

Behind The Wind - it is the atmosphere of Egyptian tombs, adventure, riddles, and running along the walls. The game was made 22.06.2021, for the Level Design module at my university, I made the game for 3 weeks.

  1. ﻿using System.Collections;
  2. using System.Collections.Generic;
  3. using UnityEngine;
  4.  
  5. public class WallRun : MonoBehaviour
  6. {
  7.     [Header("Movement")]
  8.     [SerializeField] private Transform orientation;
  9.  
  10.     [Header("Detection")]
  11.     [SerializeField] private float wallDistance = .5f;
  12.     [SerializeField] private float minimumJumpHeight = 1.5f;
  13.  
  14.     [Header("Wall Running")]
  15.     [SerializeField] private float wallRunGravity;
  16.     [SerializeField] private float wallRunJumpForce;
  17.  
  18.     [Header("Camera")]
  19.     [SerializeField] private Camera cam;
  20.     [SerializeField] private float fov;
  21.     [SerializeField] private float wallRunfov;
  22.     [SerializeField] private float wallRunfovTime;
  23.     [SerializeField] private float camTilt;
  24.     [SerializeField] private float camTiltTime;
  25.  
  26.     public float tilt { get; private set; }
  27.  
  28.     private bool wallLeft = false;
  29.     private bool wallRight = false;
  30.  
  31.     RaycastHit leftWallHit;
  32.     RaycastHit rightWallHit;
  33.  
  34.     private Rigidbody rb;
  35.  
  36.     bool CanWallRun()
  37.     {
  38.         return !Physics.Raycast(transform.position, Vector3.down, minimumJumpHeight);
  39.     }
  40.  
  41.     private void Start()
  42.     {
  43.         rb = GetComponent<Rigidbody>();
  44.     }
  45.  
  46.     void CheckWall()
  47.     {
  48.         wallLeft = Physics.Raycast(transform.position, -orientation.right, out leftWallHit, wallDistance);
  49.         wallRight = Physics.Raycast(transform.position, orientation.right, out rightWallHit, wallDistance);
  50.     }
  51.  
  52.     private void Update()
  53.     {
  54.         CheckWall();
  55.  
  56.         if (CanWallRun())
  57.         {
  58.             if (wallLeft)
  59.             {
  60.                 StartWallRun();
  61.                 Debug.Log("wall running on the left");
  62.             }
  63.             else if (wallRight)
  64.             {
  65.                 StartWallRun();
  66.                 Debug.Log("wall running on the right");
  67.             }
  68.             else
  69.             {
  70.                 StopWallRun();
  71.             }
  72.         }
  73.         else
  74.         {
  75.             StopWallRun();
  76.         }
  77.     }
  78.  
  79.     void StartWallRun()
  80.     {
  81.         rb.useGravity = false;
  82.  
  83.         rb.AddForce(Vector3.down * wallRunGravity, ForceMode.Force);
  84.  
  85.         cam.fieldOfView = Mathf.Lerp(cam.fieldOfView, wallRunfov, wallRunfovTime * Time.deltaTime);
  86.  
  87.         if (wallLeft)
  88.             tilt = Mathf.Lerp(tilt, -camTilt, camTiltTime * Time.deltaTime);
  89.         else if (wallRight)
  90.             tilt = Mathf.Lerp(tilt, camTilt, camTiltTime * Time.deltaTime);
  91.  
  92.  
  93.         if (Input.GetKeyDown(KeyCode.Space))
  94.         {
  95.             if (wallLeft)
  96.             {
  97.                 Vector3 wallRunJumpDirection = transform.up + leftWallHit.normal;
  98.                 rb.velocity = new Vector3(rb.velocity.x, 0, rb.velocity.z);
  99.                 rb.AddForce(wallRunJumpDirection * wallRunJumpForce * 100, ForceMode.Force);
  100.             }
  101.             else if (wallRight)
  102.             {
  103.                 Vector3 wallRunJumpDirection = transform.up + rightWallHit.normal;
  104.                 rb.velocity = new Vector3(rb.velocity.x, 0, rb.velocity.z); 
  105.                 rb.AddForce(wallRunJumpDirection * wallRunJumpForce * 100, ForceMode.Force);
  106.             }
  107.         }
  108.     }
  109.  
  110.     void StopWallRun()
  111.     {
  112.         rb.useGravity = true;
  113.  
  114.         cam.fieldOfView = Mathf.Lerp(cam.fieldOfView, fov, wallRunfovTime * Time.deltaTime);
  115.         tilt = Mathf.Lerp(tilt, 0, camTiltTime * Time.deltaTime);
  116.     }
  117. }
