About the project:

Game Engine: Unity

Category: Team Game

My role: Gameplay Programmer, Game Designer, Writer.

Team: Denis Yurkov, Aliaksei Yunevich, Mykhailo Baranets.

Published: itch.io

Game Was Made: February 3, 2021




Control:

W/D - Move,

Space - Ax blow,

Left Alt - Roll,

Shift - Run,

F - Read, Speak,

Esc - Menu.



Description of the game:

Viking trying to get home. It was interesting to work in a team, to share responsibilities among ourselves, most of the time I wrote code to create a gameplay, figured out how to make AI for 2d, figured out how to work with Animator, wrote dialogues for the game and how the plot would develop as a whole, created level and set the light, Lesha wrote the code for the puzzle with nuts, and he created and programmed the entire user interface for the game, and created a squirrel with pixel graphics, Misha created almost all the graphics in the game, and wrote a script for dialogues, helped me with a script to control character.

PlayerController

  1. using UnityEngine;
  2. using UnityEngine.UI;
  3. using UnityEngine.SceneManagement;
  4. using UnityEngine.Experimental.Rendering.Universal;
  5. public class PlayerController : MonoBehaviour
  6. {
  7.     [Header("Player Settings")]
  8.     public Transform attackPos;
  9.     public LayerMask WhatIsEnemies;
  10.  
  11.     public float healthPlayer;
  12.     public int damagePlayer;
  13.  
  14.     public float walkSpeed;
  15.     public float runSpeed;
  16.     public float attackRange;
  17.  
  18.     [Header("Player Animator")]
  19.     public Animator playerAnimator;
  20.  
  21.     [Header("Scene Settings")]
  22.     public string nameRestartScene;
  23.  
  24.     [Header("Light2D")]
  25.     public GameObject light2D;
  26.  
  27.     // Private Settings.
  28.     private Animator restartGameAnimation;
  29.     private Animator camAnim;
  30.     private float moveInput;
  31.     private bool bodyRight = true;
  32.     private Rigidbody2D rb;
  33.  
  34.     private void Start()
  35.     {
  36.         playerAnimator = GetComponent<Animator>();
  37.         camAnim = Camera.main.GetComponent<Animator>();
  38.         rb = GetComponent<Rigidbody2D>();
  39.         restartGameAnimation = light2D.GetComponent<Animator>();
  40.     }
  41.  
  42.     private void Update()
  43.     {
  44.         PlayerMove();
  45.         OnAttack();
  46.         DieCharacter();
  47.     }
  48.  
  49.     private void PlayerMove()
  50.     {
  51.         moveInput = Input.GetAxis("Horizontal");
  52.         playerAnimator.SetInteger("State", 0);
  53.         rb.velocity = new Vector2(moveInput * walkSpeed, rb.velocity.y);
  54.     }
  55.  
  56.     private void FixedUpdate()
  57.     {
  58.         RunAndWalkSpeed();
  59.         RoleCharacter();
  60.  
  61.         // Flip Logic!
  62.         if (bodyRight == false && moveInput > 0)
  63.         {
  64.             Flip();
  65.         }
  66.         else if (bodyRight == true && moveInput < 0)
  67.         {
  68.             Flip();
  69.         }
  70.     }
  71.  
  72.     private void DieCharacter()
  73.     {
  74.         if (healthPlayer <= 0)
  75.         {
  76.             playerAnimator.SetInteger("State", 6);
  77.             restartGameAnimation.SetTrigger("Is_Die");
  78.             Invoke("GameRestart", 2f);
  79.         }
  80.  
  81.     }
  82.     private void RunAndWalkSpeed()
  83.     {
  84.         // Walk.
  85.         if (Input.GetKey(KeyCode.A) || Input.GetKey(KeyCode.D))
  86.         {
  87.             playerAnimator.SetInteger("State", 1);
  88.         }
  89.  
  90.         // Run.
  91.         if (Input.GetKey(KeyCode.LeftShift) && (Input.GetKey(KeyCode.A) || Input.GetKey(KeyCode.D)))
  92.         {
  93.             walkSpeed = runSpeed;
  94.             playerAnimator.SetInteger("State", 2);
  95.         }
  96.  
  97.     }
  98.  
  99.     private void RoleCharacter()
  100.     {
  101.         if (Input.GetKey(KeyCode.LeftAlt))
  102.         {
  103.             playerAnimator.SetInteger("State", 3);
  104.         }
  105.     }
  106.     private void Flip()
  107.     {
  108.         bodyRight = !bodyRight;
  109.         Vector3 Scaler = transform.localScale;
  110.         Scaler.x *= -1;
  111.         transform.localScale = Scaler;
  112.     }
  113.  
  114.     private void OnAttack()
  115.     {
  116.  
  117.         if (Input.GetKeyDown(KeyCode.Space))
  118.         {
  119.             playerAnimator.SetInteger("State", 4);
  120.             camAnim.SetTrigger("cameraAnimation");
  121.  
  122.  
  123.             Collider2D[] enemies = Physics2D.OverlapCircleAll(attackPos.position, attackRange, WhatIsEnemies);
  124.             for (int i = 0; i < enemies.Length; i++)
  125.             {
  126.                 enemies[i].GetComponent<Enemy>().animatorEnemy.SetTrigger("ZombiHit");
  127.                 enemies[i].GetComponent<Enemy>().TakeDamagePlayer(damagePlayer);
  128.             }
  129.         }
  130.     }
  131.  
  132.     private void GameRestart()
  133.     {
  134.         SceneManager.LoadScene(nameRestartScene);
  135.     }
  136.  
  137.     private void OnDrawGizmosSelected()
  138.     {
  139.         Gizmos.color = Color.red;
  140.         Gizmos.DrawWireSphere(attackPos.position, attackRange);
  141.     }
  142. }

  11.     public float spawnDelay = 3f;
  12.  
  13.  
  14.     private void Start()
  15.     {
  16.         InvokeRepeating("SpamEnem", spawnDelay, spawnTime);
  17.  
  18.     }
  19.  
  20.     void SpamEnem()
  21.     {
  22.         for (int i = 0; i < 1; i++)
  23.         {
  24.             Vector2 spawnPosition = new Vector2(transformGameObject.transform.position.x , transformGameObject.transform.position.y);

  26.         }
  27.     }
  28. }

Enemy

  1. using System.Collections;
  2. using System.Collections.Generic;
  3. using UnityEngine;
  4. using Pathfinding;
  5.  
  6. public class Enemy : MonoBehaviour
  7. {
  8.     [Header("Life and Damage Enemy")]
  9.     public float healthEnemy;
  10.     public float damageEnemy;
  11.  
  12.     [Header("Animator and AI Enemy")]
  13.     public Animator animatorEnemy;
  14.     public AIPath enemyAI;
  15.  
  16.     [Header("Transform Settings Enemy")]
  17.     public float valueEnemyChangeYTransformToDie;
  18.     public Transform enemyTransform;
  19.     public BoxCollider2D boxCollider2D;
  20.  
  21.     void Start()
  22.     {
  23.         animatorEnemy = GetComponent<Animator>();
  24.     }
  25.  
  26.     void Update()
  27.     {
  28.         if (healthEnemy <= 0)
  29.         {
  30.             animatorEnemy.SetTrigger("ZombiDie");
  31.             Destroy(enemyAI);
  32.             enemyTransform.position = new Vector2(enemyTransform.position.x, valueEnemyChangeYTransformToDie);
  33.             Destroy(boxCollider2D);
  34.         }
  35.     }
  36.  
  37.     public void TakeDamagePlayer(int damagePlayer)
  38.     {
  39.         healthEnemy -= damagePlayer;
  40.     }
  41.  
  42.     private void OnTriggerEnter2D(Collider2D collision)
  43.     {
  44.         if (collision.gameObject.tag == "Player")
  45.         {
  46.             animatorEnemy.SetTrigger("ZombiAttack");
  47.             collision.GetComponent<PlayerController>().playerAnimator.SetInteger("State", 5);
  48.             collision.GetComponent<PlayerController>().healthPlayer -= damageEnemy;
  49.         }
  50.     }
  51. }
