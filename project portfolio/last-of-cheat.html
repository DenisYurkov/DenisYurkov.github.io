PlayerMovement

  1. using System.Collections;
  2. using System.Collections.Generic;
  3. using UnityEngine;
  4.  
  5. public class PlayerMovement : MonoBehaviour
  6. {
  7.     [SerializeField] CharacterController playerController;
  8.  
  9.     [SerializeField] float speedMove = 12f;
  10.     [SerializeField] float gravity = -9.81f;
  11.  
  12.     [SerializeField] Transform gravityCheck;
  13.     [SerializeField] float gravityDistance = 0.4f;
  14.     [SerializeField] LayerMask gravityMask;
  15.  
  16.     [SerializeField] float jumpHeight = 3f; 
  17.  
  18.     Vector3 velocuty;
  19.     bool isGrounded;
  20.  
  21.     // Update is called once per frame
  22.     void Update()
  23.     {
  24.         isGrounded = Physics.CheckSphere(gravityCheck.position, gravityDistance, gravityMask);
  25.  
  26.         if (isGrounded && velocuty.y < 0)
  27.         {
  28.             velocuty.y = -2f;
  29.         }
  30.  
  31.         float x = Input.GetAxis("Horizontal");
  32.         float z = Input.GetAxis("Vertical");
  33.  
  34.         Vector3 move = transform.right * x + transform.forward * z;
  35.         playerController.Move(move * speedMove * Time.deltaTime);
  36.  
  37.         if (Input.GetButtonDown("Jump") && isGrounded)
  38.         {
  39.             velocuty.y = Mathf.Sqrt(jumpHeight * -2f * gravity);
  40.         }
  41.         velocuty.y += gravity * Time.deltaTime;
  42.  
  43.         // s = 1/2 * g * t2 - Free fall formula.
  44.         playerController.Move(velocuty * Time.deltaTime);
  45.     }
  46. }

  1. using UnityEngine;
  2.  
  3. public class Gun : MonoBehaviour
  4. {
  5.     [Header("Gun")]
  6.     [SerializeField] float damage = 10f;
  7.     [SerializeField] float range = 100f;
  8.  
  9.     [Header("Camera")]
  10.     [SerializeField] Camera fpsCam;
  11.  
  12.     [Header("Particle")]
  13.     public ParticleSystem shotEffect;
  14.     [SerializeField] ParticleSystem explosionAndDieEffect;
  15.     [SerializeField] public float timeParticleDie = 1f;
  16.  
  17.     [Header("Audio")]
  18.     public AudioSource audioShot;
  19.     [SerializeField] AudioSource barrelExplosion;
  20.  
  21.     [Header("UI")]
  22.     [SerializeField] GameObject DarkBG;
  23.  
  24.     private void Start()
  25.     {
  26.         Cursor.visible = false;
  27.     }
  28.     // Update is called once per frame
  29.     void Update()
  30.     {
  31.         if (Input.GetButtonDown("Fire1"))
  32.         {
  33.             Shoot();
  34.         }
  35.     }
  36.  
  37.     private void Shoot()
  38.     {
  39.         if (DarkBG.activeSelf == true)
  40.         { 
  41.  
  42.             shotEffect.Stop();
  43.             audioShot.Stop();
  44.  
  45.         }
  46.         else {
  47.             shotEffect.Play();
  48.             audioShot.Play();
  49.  
  50.             RaycastHit hit;
  51.             if (Physics.Raycast(fpsCam.transform.position, fpsCam.transform.forward, out hit, range))
  52.             {
  53.                 Target target = hit.transform.GetComponent<Target>();
  54.  
  55.                 if (target != null)
  56.                 {
  57.                     target.TakeDamage(damage);
  58.                 }
  59.  
  60.                 if (hit.transform.gameObject.tag == "barrel roll" || hit.transform.gameObject.tag == "die effect")
  61.                 {
  62.  
  63.                     var particale = Instantiate(explosionAndDieEffect, hit.point, Quaternion.LookRotation(hit.normal));
  64.                     Destroy(particale.gameObject, timeParticleDie);
  65.                 }
  66.  
  67.                 if (hit.transform.gameObject.tag == "barrel explosion")
  68.                 {
  69.                     var particale = Instantiate(explosionAndDieEffect, hit.point, Quaternion.LookRotation(hit.normal));
  70.                     barrelExplosion.Play();
  71.                     Destroy(particale.gameObject, timeParticleDie);
  72.                 }
  73.             }
  74.         }
  75.     }
  76. }
