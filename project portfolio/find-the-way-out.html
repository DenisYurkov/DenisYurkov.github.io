Video Gameplay:


About the project:

Game Engine: Unity

Category: Game Jam

Published: itch.io

Game Was Made: March 7, 2021

Control:

W/S/D/A - Move,

Space - Jump,

E - Interaction.


Description of the game:

This is a game that combines 2D and 3D, and where you need to find a way out, the game was made in 48 hours on magejam-vol-iv.

Theam: Entire Game on One Screen.

It was great creating a game in a short time, you find yourself in a state for two whole days, first you start with a prototype, test the mechanics, and try to do everything in time everything is done quickly, but in the end, you are satisfied with yourself, you did it, you could, and you did not give up, I advise everyone to participate in GameJams.

  1. ﻿using System.Collections;
  2. using System.Collections.Generic;
  3. using UnityEngine;
  4.  
  5. public class PlayerMove : MonoBehaviour
  6. {
  7.     [Header("Settings For Move Player")]
  8.     [SerializeField] Rigidbody playerRigidbody;
  9.     [SerializeField] float speedPlayer = 6.0f;
  10.     [SerializeField] float jumpForce = 10.0f;
  11.     [SerializeField] LayerMask groundMask;
  12.  
  13.     private bool isGrounded;
  14.  
  15.     // Start is called before the first frame update
  16.     void Start()
  17.     {
  18.         playerRigidbody = GetComponent<Rigidbody>();    
  19.     }
  20.  
  21.     // Update is called once per frame
  22.     void Update()
  23.     {
  24.         // Input.
  25.         float x = Input.GetAxisRaw("Horizontal") * speedPlayer;
  26.         float y = Input.GetAxisRaw("Vertical") * speedPlayer;
  27.  
  28.         // Grounding.
  29.         isGrounded = Physics.CheckSphere(new Vector3(transform.position.x, transform.position.y - 1, transform.position.z), 0.45f, groundMask);
  30.  
  31.         // Jumping.
  32.         if (Input.GetKeyDown(KeyCode.Space) && isGrounded)
  33.         {
  34.             playerRigidbody.velocity = new Vector3(playerRigidbody.velocity.x, jumpForce, playerRigidbody.velocity.z);
  35.         }
  36.  
  37.         // Moving.
  38.         Vector3 movePos = transform.right * x + transform.forward * y;
  39.         Vector3 newMovePos = new Vector3(movePos.x, playerRigidbody.velocity.y, movePos.z);
  40.         playerRigidbody.velocity = newMovePos;
  41.  
  42.     }
  43. }
